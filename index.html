function cleanAndHighlight(text) { const map = new Map(); // 1. HTML koru text = text.replace(/<(strong|b)>.*?<\/(strong|b)>/gi, m => { const k = "__STRONG_" + map.size + "__"; map.set(k, m); return k; }); text = text.replace(/<(em|i)>.*?<\/(em|i)>/gi, m => { const k = "__EM_" + map.size + "__"; map.set(k, m); return k; }); // 2. NBSP temizle text = text.replace(/\u00A0/g, " "); // 3. whitespace normalize (AMA KELİME ARAMASINDAN ÖNCE DEĞİL) text = text.replace(/\s+/g, " ").trim(); // 4. RENKLENDİRME (EN KRİTİK DÜZELTME) const keywords = [ "yüksek kaliteli", "hakiki deri", "unisex", "birinci sınıf", "1. kalite", "1. sınıf", "1 . kalite", "1 . sınıf" ]; keywords.forEach(word => { // daha sağlam regex: kelime sınırı + boşluk toleransı const regex = new RegExp(`\\b${word.replace(/\./g, "\\.").replace(/ /g, "\\s*")}\\b`, "gi"); text = text.replace(regex, match => { return `${match}`; }); }); // 5. geri yükle map.forEach((v, k) => { text = text.replace(k, v); }); return text; } document.getElementById("input").addEventListener("input", async function () { const cleaned = cleanAndHighlight(this.value); document.getElementById("output").innerHTML = cleaned; try { await navigator.clipboard.writeText(cleaned); } catch (e) {} });
